# Python Virtual Environment (venv) Cheat Sheet


######################################
Windows
######################################


# --- Create & Activate ---
python -m venv venv              	# Create virtual env named "venv"
python3 -m venv venv             	# Same, explicitly with python3 (Linux/macOS)
venv\Scripts\activate            		# Activate env (Windows CMD)
.\venv\Scripts\activate          		# Activate env (PowerShell)
source venv/bin/activate         	# Activate env (Linux/macOS)

# --- Deactivate ---
deactivate                       # Exit virtual environment

# --- Package Management (inside venv) ---
pip install package              # Install package
pip install package==1.2.3       # Install specific version
pip uninstall package            # Remove package
pip list                         # List installed packages
pip freeze > requirements.txt    # Export env to requirements file
pip install -r requirements.txt  # Install from requirements file
pip show package                 # Show package details

# --- Upgrade pip ---
python -m pip install --upgrade pip

# --- Delete Environment ---
rm -rf venv                      # Remove venv folder (Linux/macOS)
rmdir /s venv                    # Remove venv folder (Windows CMD)
